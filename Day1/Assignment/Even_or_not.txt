namespace SecondApp
{
    internal class Program
    {
        static void Main(string[] args)
        {
            int num;
            Console.WriteLine("Please enter a number to check whether it is even or not");
            num=Convert.ToInt32(Console.ReadLine());
            if (num%2==0)
                Console.WriteLine(num + " is an even number");
            else
                Console.WriteLine($"{num} is not an even number(odd)");
        }
    }
}
