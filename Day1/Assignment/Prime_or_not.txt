namespace SecondApp
{
    internal class Program
    {
        static void Main(string[] args)
        {
            int num, flag;
            Console.WriteLine("Please enter a number to check whether it is prime or not");
            num = Convert.ToInt32(Console.ReadLine());
            if (num == 0 || num == 1)
            {
                Console.WriteLine("Neither prime not composite");
            }
            else if (num == 2 || num == 3)
            {
                Console.WriteLine($"{num} is a prime number ");
            }
            else
            {
                flag = 0;
                for (int i = 4; i <= num-1; i++)
                {
                    if (num % i != 0)
                        continue;
                    //Console.WriteLine(num + " is a prime number");
                    else
                        flag += 1;
                    //Console.WriteLine($"{num} is not a prime(composite) number");
                }
                if (flag == 0)
                {
                    Console.WriteLine($"{num} is a prime number");
                }
                else
                {
                    Console.WriteLine($"{num} is not a prime(composite) number");
                }
            }
        }
    }
}
